## Практика 
Программа для преобразования числа в строку и склейки двух строк. Разработать без использования системных функций (отдельно для Windows и Linux)

Шаг 1. Для начала напишем функцию для конкатенации 2-х строк. 

``` C
#include <malloc.h>
#include <stdio.h>
#include <stdlib.h>

char* concat(char* first, char* second) {
	char* result = (char*)calloc(100, sizeof(char));
	int size = 0;
	int index = 0;
	while (first[index] != '\0') {
		result[size++] = first[index++];
	}

	index = 0;
	while (second[index] != '\0') {
		result[size++] = second[index++];
	}
	result[size] = '\0';
	return result;
}

int main() {
	char* str = concat("hello", "world");
	printf("%s", str);
}
```

Шаг 2. Функция для преобразования числа в строку. 

``` C
#include <stdio.h>
#include <stdlib.h>

void int_to_str(int num, char* str) {
    int i = 0;
    int is_negative = 0;

    // Обработка отрицательных чисел
    if (num < 0) {
        is_negative = 1;
        num = -num;  // Делаем число положительным
    }

    // Обработка нуля
    if (num == 0) {
        str[0] = '0';
        str[1] = '\0';
        return;
    }

    // Записываем цифры в обратном порядке
    while (num > 0) {
        int digit = num % 10;          // Получаем последнюю цифру
        str[i++] = digit + '0';        // Преобразуем цифру в символ
        num /= 10;                     // Убираем последнюю цифру
    }

    // Добавляем минус для отрицательных чисел
    if (is_negative) {
        str[i++] = '-';
    }

    str[i] = '\0';  // Добавляем терминирующий ноль

    // Разворачиваем строку (так как цифры были записаны в обратном порядке)
    for (int j = 0; j < i / 2; j++) {
        char temp = str[j];
        str[j] = str[i - j - 1];
        str[i - j - 1] = temp;
    }
}

int main() {
    system("chcp 1251");
    char str[100];
    int_to_str(312, str);
    printf("Число: %d -> Строка: \"%s\"\n", 312, str);

    return 0;
}
```
