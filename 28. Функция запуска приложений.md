## Практика 

Создание функции, аргументом которой является указатель на функцию. Создать не менее двух вспомогательных функций и передать их в качестве аргумента исходной функции. Продемонстрировать специфику работы исходной функции в зависимости от переданных параметров (отдельно для Windows и Linux)

``` C
#include <stdio.h>
typedef double (*delegate)(double);

double square(double n)
{
	return n * n;
}

double cube(double n)
{
	return n * n * n;
}

double calculate(delegate func, double n) {
	if (func == NULL) return 0; // Обработка случая NULL-указателя
	return func(n);
}

int main()
{
	system("chcp 1251");
	printf("Квадрат 5: %.2f\n", calculate(square, 5));
	printf("Куб 5: %.2f\n", calculate(cube, 5));
	printf("Вызов с NULL указателем: %.2f\n", calculate(NULL, 5));
	return 0;
}
```
